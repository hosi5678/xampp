// jsonについて
javascriptのデータオブジェクト表現の一つ(連想配列と同様)
ただし、配列,連想配列とは区別される。(arrayのメソッドは不可)
連想配列は
boxes={key:"value"}(アクセスはconsole.log(boxes["key0"]);)
jsonは
boxes{"key":"value"}
で値を格納する。
(アクセスはconsole.log(boxes.key0);)
連想配列からjsonの変換は
(すなわちkey->"key"にするには)
var json=JSON.stringify(boxes);
とする。

使い方としては、
連想配列->json文字列->連想配列に戻す。
var json_arry = JSON.stringify(json_arry);
series.push(JSON.parse(json_arry));

以下、使用したコード
// 連想配列(dictionary)
    var dict={
        apple:"りんご",
        banana:"ばなな",
        cherry:"チェリー",
    };

    console.log("連想配列:"+dict["apple"]);

    // json
    var obj = {
      "apple": "リンゴ",
      "banana": "バナナ",
      "cherry": "チェリー"
    };
     
    console.log("json:"+obj.apple);

    /** データをJSON文字列に変換 */
    var json = JSON.stringify(dict);
    console.log("json stringify:"+json);

    // jsonからdictionaryへ変換
    var dict=JSON.parse(json);
    console.log("dict:"+dict);

    // object objectはdictionary,その要素を調べるには
    for(key in dict){
      console.log(key+":"+dict[key]);
    }
